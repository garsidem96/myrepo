gapminder2 <- gapminder %>%
  mutate(pop = pop/100000) %>%
  filter(country != "Kuwait")
Weighted_mean <- gapminder2 %>%
  group_by(continent, year) %>%
  summarise(wm = weighted.mean(gdpPercap, pop), pop=sum(pop))
